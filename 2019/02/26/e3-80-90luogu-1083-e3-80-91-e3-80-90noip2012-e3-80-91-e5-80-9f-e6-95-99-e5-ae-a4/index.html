二分做法

O(mlogn)O(mlogn)O(mlogn)的复杂度，常数还挺大的。。一直以为过不了，但是其实跑的挺快的。 你会发现，其实就是在枚举某个订单能否被接受，如果能，那这个点前面的都能；如果不能，那这个点后面的都不能；符合单调性,直接瞎∗∗**∗∗二分。

#include<cstdio>
#include<cstring>
#include<cmath>
#include<cstring>
#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;
const int _ = 1e6 + 100;
struct Need{
    int d;
    int s;
    int t;
}ned[_];
int A[_];
int n, m;
int C[_];
inline bool check(int k)
{
    memset(C, 0, sizeof(C));
//  printf("Id = %d : ", k);
    for(register int i = 1;i <= k;i++)
    {
        C[ned[i].s] -= ned[i].d;
        C[ned[i].t + 1] += ned[i].d;
    }
    int now = 0;
    for(register int i = 1;i <= n;i++)
    {
        now += C[i];
        if(A[i] + now < 0)
        {
//          printf("false\n");
            return false; 
        }
    } 
//  printf("true\n");
    return true;
}
int work()
{
    int le = 1, ri = n;
    int ans;
    while(le <= ri)
    {
        int mid = (le + ri) >> 1;
        if(check(mid)) ans = mid, le = mid + 1;
        else ri = mid - 1;
    }
    return ans + 1;
}
int main()
{
    scanf("%d%d", &n, &m);
    for(register int i = 1;i <= n;i++)
        scanf("%d", A + i);
    for(register int i = 1;i <= m;i++)
        scanf("%d%d%d", &ned[i].d, &ned[i].s, &ned[i].t);
    int ans = work();
    if(ans > n)
        puts("0");
    else
        printf("-1\n%d", ans);
    return 0;
}

线段树做法

O(mlogn)O(mlogn)O(mlogn)以为数据结构能跑的呲呲的，但是线段树的常数……是一个尴尬的问题。。 O(mlogn)O(mlogn)O(mlogn)的复杂度，过10510^{5}105的数据竟然困难……（吸一口O2O_2O2​才能勉强过） 在使用define定义min()时，（#define min(a, b) (a)<(b)?(a):(b)）一定不要把它使用在递归结构中，他会访问两次以上的参数，对于递归，这是致命的。 结果就是，好好的线段树跑的比暴力慢……

inline int min(int x, int y)
{
    if(x < y)
        return x;
    else
        return y;
}
struct Node{
    int MIN;
    int Sum;
    int tar;
    Node *lson, *rson;
    Node()
    {
        MIN = 0;
        tar = 0; 
    }
}POOL[_NODE];
int __tot__ = 0;
Node * malloc_() { return & POOL[__tot__++]; }
Node * null;
inline void updata(Node *tree)
{
    tree -> MIN = min(tree -> rson -> MIN, tree -> lson -> MIN);
}

inline void initNode(Node * tree)
{
    tree -> lson = tree -> rson = null;
}

inline void build(Node *tree, int l, int r)
{
    if(l == r)
    {
        scanf("%d", &tree -> MIN); 
        return ;
    }
    tree -> lson = malloc_();
    initNode(tree -> lson);
    tree -> rson = malloc_();
    initNode(tree -> rson);
    int mid = (l + r) >> 1;
    build(tree -> lson, l, mid);
    build(tree -> rson, mid + 1, r);
    updata(tree);
}

inline void tar(Node * tree, int l, int r, int val)
{
    if(tree == null) return;
    tree -> MIN -= val;
    tree -> tar += val;
}

inline void push(Node * tree, int nowl, int nowr)
{
    if(nowl == nowr || tree == null) return;
    int mid = (nowl + nowr) >> 1;
    tar(tree -> lson, nowl, mid, tree -> tar);
    tar(tree -> rson, mid + 1, nowr, tree -> tar);
    tree -> tar = 0;
}
inline void change(Node * tree, int l, int r, int val, int nowl,int nowr)
{
    if(l <= nowl && nowr <= r)
    {
        tar(tree, nowl, nowr, val);
        return ;
    }
    push(tree, nowl, nowr);
    int mid = (nowl + nowr) >> 1;
    if(l <= mid) change(tree -> lson, l, r, val, nowl, mid);
    if(r > mid) change(tree -> rson, l, r, val, mid + 1, nowr);
    updata(tree);
}

inline int query(Node * tree, int l, int r, int nowl, int nowr)
{
    if(l <= nowl && nowr <= r)
        return tree -> MIN; 
    int mid = (nowl + nowr) >> 1;
    push(tree, nowl, nowr);
    int ans = inf ;
    if(l <= mid) ans = min(ans, query(tree -> lson, l, r, nowl, mid)) ;
    if(r > mid)  ans = min(ans, query(tree -> rson, l, r, mid + 1, nowr));
    return ans;
}
