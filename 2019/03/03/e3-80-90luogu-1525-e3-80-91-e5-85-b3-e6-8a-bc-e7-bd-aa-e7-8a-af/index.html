【luogu-1525】关押罪犯

// luogu-judger-enable-o2
#include<cstdio>
#include<cstring>
#include<cmath>
#include<cstring>
#include<iostream>
#include<cmath>
#include<algorithm>
#include<map>
using namespace std;
const int _N = 21000;
const int _M = 110000;
struct Find_Union_Set{
    int F[_N << 1];
    Find_Union_Set(){}
    Find_Union_Set(int n){ for(register int i = 0;i <= n;i++) F[i] = i;}
    int find(int x)
    {
        return F[x] == x ? F[x] : F[x] = find(F[x]);
    }
    bool ask(int x, int y)
    {
        return this -> find(x) == this -> find(y) ;
    }
    void marge(int x, int y)
    {
        int f1 = this -> find(x), f2 = this -> find(y);
        if(f1 == f2) return;
        F[f1] = f2;
    }
};
Find_Union_Set set;
struct Edge{
    int a;
    int b;
    int w;
    bool operator < (const Edge & X) const {
        return w > X.w;
    }
}E[_M];
int main()
{
    int n, m;
    scanf("%d%d", &n, &m);
    set = Find_Union_Set((n << 1) + 10);
    for(register int i = 1;i <= m;i++)
        scanf("%d%d%d", &E[i].a, &E[i].b, &E[i].w);
    sort(E + 1, E + 1 + m);
    int ans = 0;
    for(register int i = 1;i <= m;i++)
    {
        int &A = E[i].a, &B = E[i].b;
        if(!set.ask(A, B))
        {
            set.marge(A, B + n);
            set.marge(A + n, B);
        }
        else
        {
            ans = E[i].w;
            break;
        }
    }
    printf("%d", ans);
    return 0;
}
