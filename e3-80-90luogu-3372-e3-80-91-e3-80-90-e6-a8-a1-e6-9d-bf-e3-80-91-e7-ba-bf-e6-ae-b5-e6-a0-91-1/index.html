#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
const int _ = 400010;
#define ll long long
struct Node{
    ll S;
    ll tar;
    Node * lson, * rson;
    Node(){ S = tar = 0; lson = rson = NULL; }
}POOL[_];
int __tot__ = 0;
inline Node * malloc_(){return &POOL[__tot__++];}
inline void updata(Node * tree) { tree -> S = tree -> lson -> S + tree -> rson -> S; }
void build(Node * tree, int l, int r)
{
    if(l == r)
    {
        scanf("%lld", &tree -> S);
        return ; 
    }
    int mid = (l + r) >> 1;
    tree -> lson = malloc_();
    tree -> rson = malloc_();
    build(tree -> lson, l, mid);
    build(tree -> rson, mid + 1, r);
    updata(tree); 
}
void tar(Node * tree, int l, int r, ll val)
{
    if(tree == NULL) return;
    tree -> S += (r - l + 1) * val;
    tree -> tar += val;
}
void push(Node * tree, int l, int r)
{
    if(tree == NULL || l == r) return;/////////////
    int mid = (l + r) >> 1;
    tar(tree -> lson, l, mid, tree -> tar);
    tar(tree -> rson, mid + 1, r, tree -> tar);
    tree -> tar = 0; 
}
void change(Node * tree, int nowl, int nowr, ll val, int l, int r)
{
    if(l <= nowl && nowr <= r)
    {
        tar(tree, nowl, nowr, val);
        return ;
    }
    push(tree, nowl, nowr);
    int mid = (nowl + nowr) >> 1;
    if(l <= mid) change(tree -> lson, nowl, mid, val, l, r);
    if(r >  mid) change(tree -> rson, mid + 1, nowr, val, l, r);
    updata(tree);
}
ll query(Node * tree, int nowl,int nowr, int l, int r)
{
    if(l <= nowl && nowr <= r)
        return tree -> S;
    int mid = (nowl + nowr) >> 1;
    ll ans = 0;
    push(tree, nowl, nowr);
    if(l <= mid) ans += query(tree -> lson, nowl, mid, l, r);
    if(r > mid)  ans += query(tree -> rson, mid + 1, nowr, l, r);
    return ans ; 
}
int main()
{
    int n, m;
    scanf("%d%d", &n, &m);
    Node * root = malloc_();
    build(root, 1, n);
    for(register int i = 1;i <= m;i++)
    {
        int op;
        scanf("%d", &op);
        if(op == 1)
        {
            int x, y;
            ll k;
            scanf("%d%d%lld", &x, &y, &k);
            change(root, 1, n, k, x, y);
        }
        else
        {
            int x, y;
            scanf("%d%d", &x, &y);
            printf("%lld\n", query(root, 1, n, x, y));
        }

     } 
    return 0;
 } 


#include<cstdio>
#include<iostream>
#include<cstring>
#include<cmath>
#define root 1,n,1
#define lson l,m,rt<<1
#define rson m+1,r,rt<<1|1
#define nownode l,r,rt
#define max(a,b) (a)>(b)?(a):(b)//虽然可以用stl或者自己写个正儿八经的函数，但是这样写起来据说能快一点，毕竟调用栈空间也是需要时间的。。。
#define min(a,b) (a)<(b)?(a):(b) //同上
#define LL long long 
using namespace std;
const int MAXN=100000+10; 
long long z[MAXN<<2];//位运算，相当于MAXN*4
long long col[MAXN<<2];//同上

void updata(int rt)//更新数据函数，rt为待更新的结点编号
{
    z[rt]=z[rt<<1]+z[rt<<1|1];//位运算，相当于z[rt/2]+z[rt/2+1]
}

void color(int l,int r,int rt,LL v)
//参数说明：区间【l，r】，编号为rt，打上“加上V”的标记
{
    z[rt]+=(r-l+1)*v;//区间修改的话，需要打lazy标记，
    //也就是需要先将此处结点（被修改区间完全覆盖的）数据更新（因为是在某一段区间的所有数都加一个数V，所以这段区间的和就要加上【区间长度】*V就好）
    col[rt]+=v;
    //为它这个结点打lazy，表示rt这个结点的孩子们都要加上V，（现在还没加上V，先打标记，不继续操作）。
    //标记颜色需要累加，不能col[rt]=v; 
}

void push_color(int l,int r,int rt)//下放标记
//参数说明：区间【l，r】，编号为rt
{
    if(col[rt]!=0)//如果这个结点有标记的话
    //（废话。。没标记还下放个pi
    {
        int m=(l+r)>>1;//相当于（l+r）/2
        color(lson,col[rt]);//为rt结点的左儿子打标记
        color(rson,col[rt]);//为rt结点的左儿子打标记
        col[rt]=0;//因为已经下放了，就清除这个结点的标记
    }
}

void build(int l,int r,int rt)//建立这个线段树
//建立区间【l，r】，编号为rt的结点
//虽然根据rt就可以算出来区间【l，r】，但是这样写会方便很多
{
    if(l==r)//已经是叶子结点了
    {
        scanf("%lld",&z[rt]);//直接读入就好
        return ;//建立完成此节点就GG
    }
    int m=(l+r)>>1;//相当于（l+r）/2
    build(lson);//建立左儿子
    build(rson);//建立右儿子
    //WARNING：以上两行顺序不可换。
    updata(rt);//俩儿子建立完成了，那就根据俩儿子的数据，来更新自己的数据就欧克。
}

void modify(int l,int r,int rt,int nowl,int nowr,LL v)//修改函数
//参数说明：在区间【l，r】（编号为rt）内为【nowl,nowr】区间加上V
//貌似很难懂
//其实就是，我们始终要修改【nowl,nowr】这个很大的区间，但是现在我们现在管不了那么多，就先修改【nowl,nowr】在【l，r】里面的部分。
{
    if(nowl<=l&&r<=nowr)//如果这个【l，r】区间已经被【nowl,nowr】完全包含了，也就是【l，r】里面的数据都要修改。
    {
        color(nownode,v);//直接打标记就欧克
        return ;
    }
    push_color(nownode);//下放标记，没错，千辛万苦打的标记就是在修改/查询的时候下放
    int m=(l+r)>>1;////相当于（l+r）/2
    if(nowl<=m)modify(lson,nowl,nowr,v);//如果需要修改左儿子，那就修改它
    if(nowr>m)modify(rson,nowl,nowr,v);//如果需要修改右儿子，那就修改它
    updata(rt);//俩儿子修改完后，更新这个结点数据
}

LL query(int l,int r,int rt,int nowl,int nowr)//终于写到查询了。。。。
//参数说明：在区间【l，r】，编号rt，查询sum（【nowl，nowr】）
//貌似又很难懂
//其实就是，我们始终要查询【nowl,nowr】这个很大的区间和，但是现在我们现在管不了那么多，就先查询【nowl,nowr】在【l，r】里面的部分数据的和。
{
    if(nowl<=l&&r<=nowr)
    {
        return z[rt];
    }
    int m=(l+r)>>1;
    LL ans=0;//WARNKING :::  VAL ans MUST INIT
    push_color(nownode);
    //下放标记，没错，千辛万苦打的标记就是在修改/查询的时候下放
    if(nowl<=m)ans+=query(lson,nowl,nowr); //如果右儿子在查询的区间里面，那就ans再加上右儿子的和
    if(nowr>m)ans+=query(rson,nowl,nowr);//如果右儿子在查询的区间里面，那就ans再加上右儿子的和 
    return ans;
 } 

int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    build(root);
    int op,x,y;LL k;
    for(int i=1;i<=m;i++)
    {
        scanf("%d",&op);
        if(op==1)
        {
            scanf("%d%d%lld",&x,&y,&k);
            modify(root,x,y,k);
        }
        else
        {
            scanf("%d%d",&x,&y);
            printf("%lld\n",query(root,x,y));
        }
    }
    return 0; //不要忘了好习惯。
}
