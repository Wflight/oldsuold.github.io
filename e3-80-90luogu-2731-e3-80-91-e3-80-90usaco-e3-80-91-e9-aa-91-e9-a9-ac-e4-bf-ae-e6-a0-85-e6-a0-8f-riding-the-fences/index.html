欧拉路径

原题 找一个奇点，开始跑。 注意DFS统计答案要在最后面把结点压入栈中。

#include<cstdio>
#include<cstring>
#include<cmath>
#include<cstring>
#include<iostream>
#include<cmath>
#include<algorithm>
#include<vector>
#include<stack>
#define max(a, b) (a)>(b)?(a):(b)
#define min(a, b) (a)<(b)?(a):(b)
#define inf 0x7fffffff
using namespace std;
const int _ = 521;
vector<int> G[_];
int vis[_][_];
stack<int>Ans;
void dfs(int k)
{
//  printf("%d\n", k);
    for(int i = 0;i < G[k].size();i++)
    {
        int exNode = G[k][i];
        if(vis[k][exNode] > 0)
        {
            vis[k][exNode]--;
            vis[exNode][k]--;
            dfs(exNode);
        }
    }
    Ans.push(k);
}
int main()
{
    memset(vis, 0, sizeof(vis));
    int F, MaxNode = -inf, MinNode = inf;
    scanf("%d", &F);
    for(register int i = 1;i <= F;i++)
    {
        int x, y;
        scanf("%d%d", &x, &y);
        vis[x][y] ++;
        vis[y][x] ++;
        G[x].push_back(y);
        G[y].push_back(x);
        MaxNode = max(MaxNode, x);
        MaxNode = max(MaxNode, y);
        MinNode = min(MinNode, x);
        MinNode = min(MinNode, y);
    }
    for(register int i = 1;i <= MaxNode;i++)
        if(G[i].size())
            sort(G[i].begin(), G[i].end());
    int S = -1;;
    for(register int i = 1;i <= MaxNode;i++)
        if(G[i].size() % 2 == 1)
        {
            S = i;
            break;
        }
    if(S != -1) dfs(S);
    else        dfs(1);
    while(!Ans.empty()) 
    {
        cout<<Ans.top()<<endl;
        Ans.pop();
    }
    return 0;
}
